### Implementation:

1. Add a state for showing particles:
```tsx
const [showParticles, setShowParticles] = useState(false);
const [particleOrigin, setParticleOrigin] = useState({ x: 0, y: 0 });
```

2. When the swipe passes threshold for approve, instead of immediately calling onApprove:
```tsx
if (offset > SWIPE_THRESHOLD) {
  dismissed.current = true;
  if (navigator.vibrate) navigator.vibrate([10, 30, 10]);
  
  // Get card center for particle origin
  const rect = cardRef.current?.getBoundingClientRect();
  if (rect) {
    setParticleOrigin({ x: rect.left + rect.width / 2, y: rect.top + rect.height / 2 });
  }
  
  // Shrink and fade the card
  await controls.start({
    scale: 0.9,
    opacity: 0,
    transition: { duration: 0.3, ease: [0.22, 1, 0.36, 1] },

  setTimeout(() => {
    onApprove(approval.id);
  }, 600);
}
```

```tsx
<AnimatePresence>

          transition={{
            duration: 0.7,
            ease: [0.22, 1, 0.36, 1],

```

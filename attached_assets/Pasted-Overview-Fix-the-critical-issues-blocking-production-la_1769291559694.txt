typescriptexport interface ApiError {
  message: string;
  code?: string;
  details?: unknown;
}

export function errorResponse(res: any, status: number, message: string, code?: string, details?: unknown) {
  const response: ApiError = { message };
  if (code) response.code = code;
  if (details && process.env.NODE_ENV !== "production") {
    response.details = details;
  }
  return res.status(status).json(response);
}

export function notFound(res: any, resource: string = "Resource") {
  return errorResponse(res, 404, `${resource} not found`, "NOT_FOUND");
}

export function badRequest(res: any, message: string, details?: unknown) {
  return errorResponse(res, 400, message, "BAD_REQUEST", details);
}

export function serverError(res: any, error: unknown, context: string) {
  console.error(`[${context}]`, error);
  return errorResponse(res, 500, "Internal server error", "INTERNAL_ERROR");
}

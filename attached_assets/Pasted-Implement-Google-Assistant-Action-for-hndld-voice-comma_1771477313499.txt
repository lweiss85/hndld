Implement Google Assistant Action for hndld voice commands.

## Overview

Create a Google Actions project using the Actions SDK that allows users to:
- Check next cleaning/appointment
- Get pending approvals
- Check spending
- Create tasks
- Control smart locks

## Server Implementation

Create server/routes/google-assistant.ts:

import { Router, Request, Response } from "express";
import {
  conversation,
  Simple,
  Card,
  Suggestion,
} from "@assistant/conversation";
import { db } from "../db";
import {
  cleaningVisits, approvals, spendingItems, tasks, smartLocks, apiTokens,
} from "@shared/schema";
import { eq, and, gte, sql, isNull } from "drizzle-orm";
import { startOfMonth, format } from "date-fns";
import { lockDoor, unlockDoor } from "../services/smart-locks";
import logger from "../lib/logger";

const router = Router();

// Google Assistant webhook handler
const app = conversation({ debug: process.env.NODE_ENV !== "production" });

// Main invocation
app.handle("main_invocation", async (conv) => {
  conv.add(new Simple({
    speech: "Welcome to hndld. You can ask about your next cleaning, pending approvals, spending, or control your smart locks. What would you like to do?",
    text: "Welcome to hndld! How can I help?",
  }));
  conv.add(new Suggestion({ title: "Next cleaning" }));
  conv.add(new Suggestion({ title: "Pending approvals" }));
  conv.add(new Suggestion({ title: "Spending summary" }));
});

// Get user context helper
async function getUserContext(conv: any): Promise<{ userId: string; householdId: string } | null> {
  const accessToken = conv.user?.params?.accessToken;
  if (!accessToken) return null;

  const [token] = await db.select().from(apiTokens)
    .where(and(
      eq(apiTokens.token, accessToken),
      isNull(apiTokens.revokedAt)
    )).limit(1);

  if (!token) return null;

  return { userId: token.userId, householdId: token.householdId };
}

// Next cleaning intent
app.handle("next_cleaning", async (conv) => {
  const context = await getUserContext(conv);
  if (!context) {
    conv.add("Please link your hndld account in the Google Home app first.");
    conv.scene.next = { name: "actions.scene.END_CONVERSATION" };
    return;
  }

  const [nextVisit] = await db.select().from(cleaningVisits)
    .where(and(
      eq(cleaningVisits.householdId, context.householdId),
      gte(cleaningVisits.scheduledAt, new Date())
    ))
    .orderBy(cleaningVisits.scheduledAt)
    .limit(1);

  if (!nextVisit) {
    conv.add("You don't have any upcoming cleaning visits scheduled.");
  } else {
    const date = format(nextVisit.scheduledAt, "EEEE, MMMM do");
    const time = format(nextVisit.scheduledAt, "h:mm a");
    conv.add(`Your next cleaning is scheduled for ${date} at ${time}.`);
  }
});

// Pending approvals intent
app.handle("pending_approvals", async (conv) => {
  const context = await getUserContext(conv);
  if (!context) {
    conv.add("Please link your hndld account first.");
    return;
  }

  const pending = await db.select({ count: sql<number>`count(*)` })
    .from(approvals)
    .where(and(
      eq(approvals.householdId, context.householdId),
      eq(approvals.status, "PENDING")
    ));

  const count = pending[0]?.count || 0;

  if (count === 0) {
    conv.add("You have no pending approvals. Everything is up to date!");
  } else {
    conv.add(`You have ${count} pending approval${count === 1 ? "" : "s"}. Would you like me to approve them all?`);
    conv.add(new Suggestion({ title: "Approve all" }));
    conv.add(new Suggestion({ title: "No thanks" }));
  }
});

// Approve all intent
app.handle("approve_all", async (conv) => {
  const context = await getUserContext(conv);
  if (!context) {
    conv.add("Please link your hndld account first.");
    return;
  }

  await db.update(approvals)
    .set({
      status: "APPROVED",
      resolvedAt: new Date(),
      resolvedBy: context.userId,
      resolution: "Approved via Google Assistant",
    })
    .where(and(
      eq(approvals.householdId, context.householdId),
      eq(approvals.status, "PENDING")
    ));

  conv.add("Done! I've approved all pending items.");
});

// Spending summary intent
app.handle("spending_summary", async (conv) => {
  const context = await getUserContext(conv);
  if (!context) {
    conv.add("Please link your hndld account first.");
    return;
  }

  const startDate = startOfMonth(new Date());

  const spending = await db.select({
    total: sql<number>`sum(${spendingItems.amountInCents})`,
    count: sql<number>`count(*)`,
  })
    .from(spendingItems)
    .where(and(
      eq(spendingItems.householdId, context.householdId),
      gte(spendingItems.date, startDate)
    ));

  const total = (spending[0]?.total || 0) / 100;
  const count = spending[0]?.count || 0;

  if (count === 0) {
    conv.add("You haven't recorded any spending this month.");
  } else {
    conv.add(`This month, you've spent $${total.toFixed(2)} across ${count} transactions.`);
  }
});

// Create task intent
app.handle("create_task", async (conv) => {
  const context = await getUserContext(conv);
  if (!context) {
    conv.add("Please link your hndld account first.");
    return;
  }

  const taskTitle = conv.intent.params?.task_title?.resolved;

  if (!taskTitle) {
    conv.add("What task would you like to create?");
    conv.scene.next = { name: "collect_task_title" };
    return;
  }

  await db.insert(tasks).values({
    householdId: context.householdId,
    title: taskTitle,
    status: "TODO",
    priority: "MEDIUM",
    createdBy: context.userId,
    source: "GOOGLE_ASSISTANT",
  });

  conv.add(`Done! I've created a task: ${taskTitle}.`);
});

// Lock door intent
app.handle("lock_door", async (conv) => {
  const context = await getUserContext(conv);
  if (!context) {
    conv.add("Please link your hndld account first.");
    return;
  }

  const doorName = conv.intent.params?.door_name?.resolved || "front door";

  const [lock] = await db.select().from(smartLocks)
    .where(and(
      eq(smartLocks.householdId, context.householdId),
      sql`lower(${smartLocks.name}) like ${`%${doorName.toLowerCase()}%`}`
    ))
    .limit(1);

  if (!lock) {
    conv.add(`I couldn't find a lock called ${doorName}.`);
    return;
  }

  const success = await lockDoor({
    lockId: lock.id,
    externalId: lock.externalId!,
    accessToken: lock.accessToken!,
  });

  if (success) {
    conv.add(`The ${lock.name} is now locked.`);
  } else {
    conv.add(`I couldn't lock the ${lock.name}. Please try again.`);
  }
});

// Unlock door intent
app.handle("unlock_door", async (conv) => {
  const context = await getUserContext(conv);
  if (!context) {
    conv.add("Please link your hndld account first.");
    return;
  }

  const doorName = conv.intent.params?.door_name?.resolved || "front door";

  const [lock] = await db.select().from(smartLocks)
    .where(and(
      eq(smartLocks.householdId, context.householdId),
      sql`lower(${smartLocks.name}) like ${`%${doorName.toLowerCase()}%`}`
    ))
    .limit(1);

  if (!lock) {
    conv.add(`I couldn't find a lock called ${doorName}.`);
    return;
  }

  const success = await unlockDoor({
    lockId: lock.id,
    externalId: lock.externalId!,
    accessToken: lock.accessToken!,
  });

  if (success) {
    conv.add(`The ${lock.name} is now unlocked.`);
  } else {
    conv.add(`I couldn't unlock the ${lock.name}. Please try again.`);
  }
});

// Fallback
app.handle("fallback", (conv) => {
  conv.add("I'm not sure how to help with that. You can ask about your next cleaning, pending approvals, or spending.");
  conv.add(new Suggestion({ title: "Next cleaning" }));
  conv.add(new Suggestion({ title: "Help" }));
});

// Express route handler
router.post("/google-assistant", app);

export function registerGoogleAssistantRoutes(appRouter: Router) {
  appRouter.use(router);
}

## Actions Console Configuration

Create actions.yaml for Google Actions Console:

actions:
  - name: actions.intent.MAIN
    fulfillment:
      webhook:
        handler: main_invocation

  - name: next_cleaning
    trainingPhrases:
      - when is my next cleaning
      - next cleaning appointment
      - when is the cleaner coming
    fulfillment:
      webhook:
        handler: next_cleaning

  - name: pending_approvals
    trainingPhrases:
      - how many approvals are pending
      - check my approvals
      - pending approvals
    fulfillment:
      webhook:
        handler: pending_approvals

  - name: approve_all
    trainingPhrases:
      - approve all
      - approve everything
      - approve them all
    fulfillment:
      webhook:
        handler: approve_all

  - name: spending_summary
    trainingPhrases:
      - how much have I spent
      - spending summary
      - what's my spending this month
    fulfillment:
      webhook:
        handler: spending_summary

  - name: create_task
    trainingPhrases:
      - create a task
      - add task $task_title
      - remind me to $task_title
    parameters:
      - name: task_title
        type: actions.type.FreeText
    fulfillment:
      webhook:
        handler: create_task

  - name: lock_door
    trainingPhrases:
      - lock the door
      - lock the $door_name
    parameters:
      - name: door_name
        type: actions.type.FreeText
    fulfillment:
      webhook:
        handler: lock_door

  - name: unlock_door
    trainingPhrases:
      - unlock the door
      - unlock the $door_name
    parameters:
      - name: door_name
        type: actions.type.FreeText
    fulfillment:
      webhook:
        handler: unlock_door

## Account Linking

Configure OAuth2 in Google Actions Console:
- Linking type: OAuth
- Grant type: Authorization code
- Client ID: (generate for Google)
- Authorization URL: https://your-domain.com/api/v1/oauth/authorize
- Token URL: https://your-domain.com/api/v1/oauth/token

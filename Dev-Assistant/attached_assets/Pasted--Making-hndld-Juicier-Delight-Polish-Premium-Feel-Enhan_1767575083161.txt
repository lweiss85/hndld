```tsx
import { useEffect, useState } from 'react';

function AnimatedNumber({ value, duration = 500 }: { value: number; duration?: number }) {
  const [count, setCount] = useState(0);
  
  useEffect(() => {
    const start = count;
    const end = value;
    const range = end - start;
    const startTime = Date.now();
    
    const timer = setInterval(() => {
      const elapsed = Date.now() - startTime;
      const progress = Math.min(elapsed / duration, 1);
      
      // Ease out cubic
      const eased = 1 - Math.pow(1 - progress, 3);
      setCount(Math.floor(start + range * eased));
      
      if (progress === 1) {
        clearInterval(timer);
      }
    }, 16);
    
    return () => clearInterval(timer);
  }, [value]);
  
  return <span>{count}</span>;
}
```

### 2.2 Pull-to-Refresh Animation

**What's missing:** No pull-to-refresh

**Add:**
```tsx
import { useState, useRef } from 'react';

export function PullToRefresh({ onRefresh, children }: { 
  onRefresh: () => Promise<void>; 
  children: React.ReactNode 
}) {
  const [pulling, setPulling] = useState(false);
  const [pullDistance, setPullDistance] = useState(0);
  const [refreshing, setRefreshing] = useState(false);
  const startY = useRef(0);
  
  const handleTouchStart = (e: React.TouchEvent) => {
    if (window.scrollY === 0) {
      startY.current = e.touches[0].clientY;
    }
  };
  
  const handleTouchMove = (e: React.TouchEvent) => {
    if (startY.current === 0 || refreshing) return;
    
    const currentY = e.touches[0].clientY;
    const distance = Math.max(0, Math.min(currentY - startY.current, 80));
    
    if (distance > 0) {
      setPulling(true);
      setPullDistance(distance);
    }
  };
  
  const handleTouchEnd = async () => {
    if (pullDistance > 60 && !refreshing) {
      setRefreshing(true);
      await onRefresh();
      setRefreshing(false);
    }
    
    setPulling(false);
    setPullDistance(0);
    startY.current = 0;
  };
  
  return (
    <div 
      onTouchStart={handleTouchStart}
      onTouchMove={handleTouchMove}
      onTouchEnd={handleTouchEnd}
    >
      {/* Pull indicator */}
      <div 
        className="fixed top-0 left-0 right-0 flex justify-center pt-4 transition-all duration-200 z-50"
        style={{
          transform: `translateY(${pulling ? pullDistance - 80 : -80}px)`,
          opacity: pullDistance / 80,
        }}
      >
        <div className="bg-card/95 backdrop-blur-lg rounded-full px-4 py-2 shadow-lg border border-border">
          {refreshing ? (
            <div className="flex items-center gap-2">
              <div className="w-4 h-4 border-2 border-primary border-t-transparent rounded-full animate-spin" />
              <span className="text-sm font-medium">Refreshing...</span>
            </div>
          ) : (
            <span className="text-sm font-medium">
              {pullDistance > 60 ? 'Release to refresh' : 'Pull to refresh'}
            </span>
          )}
        </div>
      </div>
      
      {children}
    </div>
  );
}

// Use on This Week page
<PullToRefresh onRefresh={async () => {
  await queryClient.invalidateQueries(['/api/dashboard']);
}}>
  {/* existing content */}
</PullToRefresh>
```

**Effect:** Refreshing feels native and delightful

```tsx
            src={preview}
```

```tsx
export function PageTransition({ children }: { children: React.ReactNode }) {
  const [location] = useLocation();
  const [displayLocation, setDisplayLocation] = useState(location);
  const [transitioning, setTransitioning] = useState(false);
```

### 3.4 Smart Scroll Reveal

**Add intersection observer animations:**
```tsx
function useScrollReveal() {
  const ref = useRef<HTMLDivElement>(null);
  const [isVisible, setIsVisible] = useState(false);
  
  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsVisible(true);
          // Once visible, stay visible
        }
      },
      { threshold: 0.1, rootMargin: '0px 0px -100px 0px' }
    );
    
    if (ref.current) {
      observer.observe(ref.current);
    }
    
    return () => observer.disconnect();
  }, []);
  
  return { ref, isVisible };
}

// Use in sections
function PrioritiesSection() {
  const { ref, isVisible } = useScrollReveal();
  
  return (
    <section 
      ref={ref}
      className={cn(
        "transition-all duration-700",
        isVisible 
          ? "opacity-100 translate-y-0" 
          : "opacity-0 translate-y-10"
      )}
    >
      {/* content */}
    </section>
  );
}
```

```tsx
function DetailModal({ onClose }: { onClose: () => void }) {
```

## TIER 6: Performance = Juice üöÄ

### 6.1 Optimistic UI Updates

**Don't wait for server:**
```tsx
const mutation = useMutation({
  mutationFn: approveApproval,
  onMutate: async (approvalId) => {
    // Cancel outgoing refetches
    await queryClient.cancelQueries(['/api/approvals']);
    
    // Snapshot previous value
    const previous = queryClient.getQueryData(['/api/approvals']);
    
    // Optimistically update
    queryClient.setQueryData(['/api/approvals'], (old) =>
      old?.map(a => 
        a.id === approvalId 
          ? { ...a, status: 'APPROVED' }
          : a
      )
    );
    
    // Return context
    return { previous };
  },
  onError: (err, variables, context) => {
    // Rollback on error
    queryClient.setQueryData(['/api/approvals'], context.previous);
```

### 6.2 Image Loading with Blur-Up

**Never show broken/loading images:**
```tsx
function BlurImage({ src, alt, className }: { src: string; alt: string; className?: string }) {
  const [loaded, setLoaded] = useState(false);
  
  return (
    <div className="relative overflow-hidden">
      {/* Blurred placeholder */}
      <img
        src={`${src}?w=20`} // Tiny version
        alt={alt}
        className={cn(
          "absolute inset-0 w-full h-full object-cover blur-2xl scale-110",
          loaded && "opacity-0 transition-opacity duration-500"
        )}
      />
      
      {/* Full image */}
      <img
        src={src}
        alt={alt}
        className={cn(
          "relative w-full h-full object-cover transition-opacity duration-500",
          loaded ? "opacity-100" : "opacity-0",
          className
        )}
        onLoad={() => setLoaded(true)}
      />
    </div>
  );
}
```

**Effect:** Images load beautifully

### Sound Helper
```tsx
// utils/sounds.ts
const cache = new Map<string, HTMLAudioElement>();

export function playSound(name: string, volume = 0.3) {
  if (localStorage.getItem('soundsEnabled') === 'false') return;
  
  let sound = cache.get(name);
  if (!sound) {
    sound = new Audio(`/sounds/${name}.mp3`);
    sound.volume = volume;
    cache.set(name, sound);
  }
  
  sound.currentTime = 0;
  sound.play().catch(() => {});
}
```

### Success Celebration
```tsx
// components/celebrate.tsx
import confetti from 'canvas-confetti';
import { playSound } from '@/utils/sounds';
import { haptic } from '@/utils/haptics';

export function celebrate(type: 'small' | 'medium' | 'large' = 'medium') {
  const configs = {
    small: { particleCount: 30, spread: 40 },
    medium: { particleCount: 60, spread: 70 },
    large: { particleCount: 100, spread: 100 },
  };
  
  confetti({
    ...configs[type],
    colors: ['#1D2A44', '#E7D8B1', '#2E7D5B'],
    origin: { y: 0.6 },
    disableForReducedMotion: true,
  });
  
  playSound('success');
  haptic.success();
}
```
